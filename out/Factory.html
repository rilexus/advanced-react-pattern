const components = { ["input"]: Input, ["checkbox"]: Checkbox, ["number"]: NumberInput, }; const DynamicComponent = ({ name, withProps }) => { if (!!components[name]) { const Component = components[name]; return <Component {...withProps} />; } return null; };
