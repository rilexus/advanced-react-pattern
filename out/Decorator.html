function EventLoggerDecorator({ children }) { return React.Children.map(children, (child) => { const childrenOwnProps = child.props; const newProps = {}; // NOTE: you can use the "reduce" function here. Object.entries(childrenOwnProps).forEach(([propKey, propValue]) => { // Intercept all event handlers. if (propKey.startsWith("on")) { newProps[propKey] = (...args) => { // Log to the console. console.log(`[EventLogger]: ${propKey}`); // Call child event handler. propValue(...args); }; } }); return React.cloneElement(child, newProps); }); }
